362 struct SceneNodeCore::getByPath_Helper<BaseSimObjectAspect&> {
363 static BaseSimObjectAspect& get(std::shared_ptr<SceneNodeCore> rootNode, const std::string& where) {
364 std::string::const_iterator div {std::find(where.begin(), where.end(), '@')};
365 assert(div != where.end() && "Must contain @ to be a valid path to a scene node's aspect");
366
367 // extract the node path from full path
368 const std::string nodePath { where.substr(0, div - where.begin()) };
369 const std::string aspectName { where.substr(1 + (div - where.begin())) };
370 assert(rootNode->getWorld().lock()->getSystem<SimSystem>()->aspectRegistered(aspectName) && "No aspect of this type has been registered with the Sim System");
371
372 std::shared_ptr<SimObject> node { rootNode->getByPath<std::shared_ptr<SimObject>>(nodePath) };
373 return node->getAspect(aspectName);
374 }
375
376 static constexpr bool s_valid { true };
377 };
