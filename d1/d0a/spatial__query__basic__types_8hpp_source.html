40 struct VolumeBase_ {
41 inline static constexpr std::array<glm::vec3, 8> GetCornerSignsArray() {
42 std::array<glm::vec3, 8> cornerSigns {};
43 for(uint8_t corner {0}; corner < 8; ++corner) {
44 cornerSigns[corner].x = corner&BoxCornerSpecifier::RIGHT? 1.f: -1.f;
45 cornerSigns[corner].y = corner&BoxCornerSpecifier::TOP? 1.f: -1.f;
46 cornerSigns[corner].z = corner&BoxCornerSpecifier::FRONT? 1.f: -1.f;
47 }
48 return cornerSigns;
49 }

51 inline static std::array<glm::vec3, 8> ComputeBoxCorners(const glm::vec3& boxDimensions) {
52 const std::array<glm::vec3, 8> cornerSignsArray { GetCornerSignsArray() };
53 glm::vec3 absoluteCornerOffset { .5f * boxDimensions };
54 std::array<glm::vec3, 8> cornerArray { .5f * boxDimensions };
55 for(uint8_t corner{0}; corner < 8; ++corner) {
56 cornerArray[corner] = cornerSignsArray[corner] * absoluteCornerOffset;
57 }
58 return cornerArray;
59 }
